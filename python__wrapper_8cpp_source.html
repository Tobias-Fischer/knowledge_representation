192  template <typename Container>
193  static void construct(PyObject* object, boost::python::converter::rvalue_from_python_stage1_data* data)
194  {
195  namespace python = boost::python;
196  // Object is a borrowed reference, so create a handle indicting it is
197  // borrowed for proper reference counting.
198  python::handle<> handle(python::borrowed(object));
199 
200  // Obtain a handle to the memory block that the converter has allocated
201  // for the C++ type.
202  typedef python::converter::rvalue_from_python_storage<Container> storage_type;
203  void* storage = reinterpret_cast<storage_type*>(data)->storage.bytes;
204 
205  typedef python::stl_input_iterator<typename Container::value_type> iterator;
206 
207  // Allocate the C++ type into the converter's memory block, and assign
208  // its handle to the converter's convertible variable. The C++
209  // container is populated by passing the begin and end iterators of
210  // the python object to the container's constructor.
211  new (storage) Container(iterator(python::object(handle)), // begin
212  iterator()); // end
213  data->convertible = storage;
214  }
